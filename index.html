<script>
// 初始化场景、相机、渲染器
const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, window.innerWidth * 0.6 / window.innerHeight, 0.1, 1000);
camera.position.z = 5;
const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(window.innerWidth * 0.6, window.innerHeight);
document.getElementById('sceneContainer').appendChild(renderer.domElement);
// 生成圣诞树粒子
const particleCount = 10000;
const geometry = new THREE.BufferGeometry();
const positions = new Float32Array(particleCount * 3);
const colors = new Float32Array(particleCount * 3);
// 圣诞树形状参数
const radius = 3;
const height = 5;
// 生成圣诞树粒子位置和金色系颜色
for (let i = 0; i < particleCount; i++) {
const u = Math.random();
const v = Math.random();
const theta = v * Math.PI * 2;
const r = radius * Math.sqrt(u);
const y = height * (u - 0.5);
positions[i * 3] = r * Math.cos(theta);
positions[i * 3 + 1] = y;
positions[i * 3 + 2] = r * Math.sin(theta);
// 金色渐变
const goldFactor = (y + height / 2) / height;
colors[i * 3] = 1.0;
colors[i * 3 + 1] = 0.8 + goldFactor * 0.2;
colors[i * 3 + 2] = 0.2 + goldFactor * 0.3;
}
geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
// 粒子材质
const material = new THREE.PointsMaterial({
size: 0.05,
vertexColors: true,
transparent: true,
opacity: 0.8
});
const tree = new THREE.Points(geometry, material);
scene.add(tree);
// 轨道控制器
const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
// 烟花效果
const fireworks = [];
function createFirework() {
const fireworkGeo = new THREE.BufferGeometry();
const fireworkPos = new Float32Array(500 * 3);
const fireworkCol = new Float32Array(500 * 3);
for (let i = 0; i < 500; i++) {
const angle = Math.random() * Math.PI * 2;
const speed = Math.random() * 0.1 + 0.05;
const life = Math.random() * 30 + 20;
const color = new THREE.Color(Math.random() * 0.5 + 0.5, Math.random() * 0.5 + 0.5, Math.random());
fireworkPos[i * 3] = Math.random() * 2 - 1;
fireworkPos[i * 3 + 1] = Math.random() * 2 + 3;
fireworkPos[i * 3 + 2] = Math.random() * 2 - 1;
fireworkCol[i * 3] = color.r;
fireworkCol[i * 3 + 1] = color.g;
fireworkCol[i * 3 + 2] = color.b;
fireworks.push({
pos: fireworkPos,
col: fireworkCol,
angle,
speed,
life,
age: 0,
geo: fireworkGeo.setAttribute('position', new THREE.BufferAttribute(fireworkPos, 3)).setAttribute('color', new THREE.BufferAttribute(fireworkCol, 3)),
mat: new THREE.PointsMaterial({ size: 0.1, vertexColors: true }),
mesh: new THREE.Points(fireworkGeo, new THREE.PointsMaterial({ size: 0.1, vertexColors: true }))
});
scene.add(fireworks[fireworks.length - 1].mesh);
}
}
function updateFireworks() {
for (let i = fireworks.length - 1; i >= 0; i--) {
const fw = fireworks[i];
fw.age++;
for (let j = 0; j < 500; j++) {
const idx = j * 3;
fw.pos[idx] += Math.cos(fw.angle) * fw.speed;
fw.pos[idx + 1] += Math.sin(fw.angle) * fw.speed;
fw.pos[idx + 2] += (Math.random() - 0.5) * 0.1;
}
fw.geo.attributes.position.needsUpdate = true;
if (fw.age > fw.life) {
scene.remove(fw.mesh);
fireworks.splice(i, 1);
}
}
if (Math.random() > 0.95) {
createFirework();
}
}
// 窗口大小变化
window.addEventListener('resize', () => {
camera.aspect = window.innerWidth * 0.6 / window.innerHeight;
camera.updateProjectionMatrix();
renderer.setSize(window.innerWidth * 0.6, window.innerHeight);
});
// 动画循环
function animate() {
requestAnimationFrame(animate);
controls.update();
updateFireworks();
renderer.render(scene, camera);
// 延迟显示文字
if (Date.now() - start > 8000) {
document.getElementById('textOverlay').classList.add('show');
}
}
const start = Date.now();
animate();
</script>
